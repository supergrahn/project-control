# Task Flow — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add pipeline guardrails, dependencies, audit logging, and priority ordering to the task system.

**Architecture:** Status transitions are validated against a deterministic state machine (idea → speccing → planning → developing → done) with soft-block warnings for missing files or unfinished dependencies. Every status change is logged to `task_status_log` with context (who, when, why). Dependencies between tasks are stored in `task_dependencies` and shown in PropertiesPanel. Task lists are sorted by priority (urgent → high → medium → low), then recency.

**Tech Stack:** Next.js App Router, better-sqlite3, Vitest + @testing-library/react, design system components from `components/ui/`

import { Database } from 'better-sqlite3'
import { Task, TaskStatus } from './tasks'
import { isTaskBlocked } from './taskDependencies'

export const VALID_TRANSITIONS: Record<TaskStatus, TaskStatus[]> = {
  'idea': ['speccing', 'done'],
  'speccing': ['planning', 'idea', 'done'],
  'planning': ['developing', 'speccing', 'done'],
  'developing': ['done', 'planning'],
  'done': ['developing']
}

export function isValidTransition(fromStatus: TaskStatus, toStatus: TaskStatus): boolean {
  return VALID_TRANSITIONS[fromStatus].includes(toStatus)
}

export function isBackwardTransition(fromStatus: TaskStatus, toStatus: TaskStatus): boolean {
  const forward = ['idea', 'speccing', 'planning', 'developing', 'done']
  return forward.indexOf(toStatus) < forward.indexOf(fromStatus)
}

export function checkReadiness(db: Database, task: Task, newStatus: TaskStatus): string[] {
  const warnings: string[] = []

  if (newStatus === 'planning' && !task.spec_file) {
    warnings.push('Spec file not set. Moving to Planning anyway.')
  }
  if (newStatus === 'developing' && !task.plan_file) {
    warnings.push('Plan file not set. Moving to Developing anyway.')
  }

  // Check unfinished dependencies
  const { getTaskDependencies } = require('./taskDependencies')
  const blockedBy = getTaskDependencies(db, task.id, 'incoming')
  const { getTask } = require('./tasks')
  const unfinished = blockedBy.filter((dep: any) => {
    const depTask = getTask(db, dep.depends_on_id)
    return depTask && depTask.status !== 'done'
  })
  if (unfinished.length > 0) {
    warnings.push(
      `${unfinished.length} unfinished ${unfinished.length === 1 ? 'dependency' : 'dependencies'}. Proceeding anyway.`
    )
  }

  return warnings
}
```

import { isValidTransition, checkReadiness } from './statusValidation'
import { logStatusChange } from './taskStatusLog'

export function transitionTaskStatus(
  db: Database,
  id: string,
  newStatus: TaskStatus,
  reason?: string
): { task: Task; warnings: string[] } {
  const task = getTask(db, id)
  if (!task) throw new Error(`Task ${id} not found`)

  if (!isValidTransition(task.status, newStatus)) {
    throw new Error(
      `Invalid transition: ${task.status} → ${newStatus}`
    )
  }

  const warnings = checkReadiness(db, task, newStatus)

  const now = new Date().toISOString()
  db.prepare('UPDATE tasks SET status = ?, updated_at = ? WHERE id = ?')
    .run(newStatus, now, id)

  logStatusChange(db, id, task.status, newStatus, 'user', reason)

  return {
    task: getTask(db, id)!,
    warnings
  }
}
```

```typescript
import { Database } from 'better-sqlite3'
import { randomUUID } from 'crypto'

export type TaskDependency = {
  id: string
  task_id: string
  depends_on_id: string
  created_at: string
}

export function addDependency(db: Database, taskId: string, dependsOnId: string): TaskDependency {
  const id = randomUUID()
  const now = new Date().toISOString()
  
  // INSERT OR IGNORE handles duplicate constraint
  db.prepare(`
    INSERT OR IGNORE INTO task_dependencies (id, task_id, depends_on_id, created_at)
    VALUES (?, ?, ?, ?)
  `).run(id, taskId, dependsOnId, now)

  // Fetch and return the record (either newly inserted or existing)
  return db.prepare(
    'SELECT * FROM task_dependencies WHERE task_id = ? AND depends_on_id = ?'
  ).get(taskId, dependsOnId) as TaskDependency
}

export function removeDependency(db: Database, taskId: string, dependsOnId: string): void {
  db.prepare('DELETE FROM task_dependencies WHERE task_id = ? AND depends_on_id = ?')
    .run(taskId, dependsOnId)
}

export function getTaskDependencies(
  db: Database,
  taskId: string,
  direction: 'incoming' | 'outgoing'
): TaskDependency[] {
  if (direction === 'incoming') {
    // Tasks this task depends on
    return db.prepare(
      'SELECT * FROM task_dependencies WHERE task_id = ? ORDER BY created_at DESC'
    ).all(taskId) as TaskDependency[]
  } else {
    // Tasks that depend on this task
    return db.prepare(
      'SELECT * FROM task_dependencies WHERE depends_on_id = ? ORDER BY created_at DESC'
    ).all(taskId) as TaskDependency[]
  }
}

export function isTaskBlocked(db: Database, taskId: string): boolean {
  const blocked = db.prepare(
    `SELECT COUNT(*) as count FROM task_dependencies
     WHERE task_id = ? AND depends_on_id IN (
       SELECT id FROM tasks WHERE status != 'done'
     )`
  ).get(taskId) as { count: number }
  return blocked.count > 0
}

export function getBlockedByTasks(db: Database, taskId: string): string[] {
  const deps = getTaskDependencies(db, taskId, 'incoming')
  return deps
    .filter(dep => {
      const depTask = db.prepare('SELECT status FROM tasks WHERE id = ?').get(dep.depends_on_id) as { status: string } | undefined
      return depTask && depTask.status !== 'done'
    })
    .map(dep => dep.depends_on_id)
}

export function getBlockingTasks(db: Database, taskId: string): string[] {
  const deps = getTaskDependencies(db, taskId, 'outgoing')
  return deps.map(dep => dep.task_id)
}
```

'use client'

import { TaskStatusLogEntry } from '@/lib/db/taskStatusLog'
import { Badge } from '@/components/ui/Badge'

interface StatusTimelineProps {
  entries: TaskStatusLogEntry[]
}

export function StatusTimeline({ entries }: StatusTimelineProps) {
  if (entries.length === 0) {
    return <p className="text-text-muted text-sm">No status changes yet</p>
  }

  const formatDate = (dateString: string): string => {
    const date = new Date(dateString)
    const now = new Date()
    const diffMs = now.getTime() - date.getTime()
    const diffDays = Math.floor(diffMs / (1000 * 60 * 60 * 24))

    if (diffDays === 0) {
      return date.toLocaleTimeString('en-US', { hour: '2-digit', minute: '2-digit' })
    } else if (diffDays === 1) {
      return 'Yesterday'
    } else if (diffDays < 7) {
      return `${diffDays} days ago`
    } else {
      return date.toLocaleDateString('en-US', { month: 'short', day: 'numeric' })
    }
  }

  const getChangedByLabel = (changedBy: string): string => {
    switch (changedBy) {
      case 'user':
        return 'user'
      case 'sync':
        return 'sync'
      case 'session':
        return 'session'
      default:
        return changedBy
    }
  }

  return (
    <div className="space-y-3">
      <h3 className="text-sm font-semibold text-text-primary">Status Changes</h3>
      <div className="space-y-2">
        {entries.map((entry) => (
          <div key={entry.id} className="flex items-start gap-3 text-sm">
            <div className="flex-1">
              <div className="text-text-primary">
                <span className="text-text-secondary">{formatDate(entry.created_at)}</span>
                {' • '}
                <span>{entry.from_status}</span>
                <span className="text-text-muted mx-1">→</span>

                {entry.reason && ` • ${entry.reason}`}
              </div>
            </div>
          </div>
        ))}

  )
}

  const validTransitions = VALID_TRANSITIONS[task.status]
  
  const handleStatusChange = async (newStatus: TaskStatus) => {
    // Get warnings from validation
    const warnings = checkReadiness(db, task, newStatus)
    
    // Open confirmation dialog
    setConfirmDialog({
      targetStatus: newStatus,
      warnings
    })
  }

  const handleConfirm = async () => {
    if (!confirmDialog) return
    
    try {
      const response = await fetch(`/api/tasks/${task.id}`, {
        method: 'PATCH',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          status: confirmDialog.targetStatus,
          reason: 'User changed status'
        })
      })
      
      if (!response.ok) throw new Error(await response.text())
      
      const updated = await response.json()
      onTaskUpdate(updated)
      setConfirmDialog(null)
    } catch (error) {
      // Handle error
      console.error('Status change failed:', error)
    }
  }

  return (
    <>
      <div className="space-y-4">
        {/* Status Dropdown */}
        <div>
          <label className="text-xs font-medium text-text-secondary">Status</label>
          <select
            value={task.status}
            onChange={(e) => handleStatusChange(e.target.value as TaskStatus)}
            className="w-full mt-1 rounded border border-border-default bg-bg-secondary text-text-primary"
          >
            {validTransitions.map(status => {
              const isBackward = isBackwardTransition(task.status, status)
              return (
                <option key={status} value={status}>
                  {status.charAt(0).toUpperCase() + status.slice(1)}
                  {isBackward ? ' (backward)' : ''}
                </option>
              )
            })}
          </select>
        </div>

        {/* Rest of panel... */}
      </div>

      {confirmDialog && (
        <StatusConfirmationDialog
          isOpen={true}
          task={task}
          targetStatus={confirmDialog.targetStatus}
          warnings={confirmDialog.warnings}
          isBlocked={false}
          blockedByTasks={[]}
          onConfirm={handleConfirm}
          onCancel={() => setConfirmDialog(null)}
        />
      )}
    </>
  )
}

import { getDb } from '@/lib/db'
import { getTask } from '@/lib/db/tasks'
import { getTaskDependencies } from '@/lib/db/taskDependencies'
import { NextResponse } from 'next/server'

export async function GET(
  req: Request,
  { params }: { params: { id: string } }
) {
  try {
    const db = getDb()
    const projectId = getProjectIdFromSession()
    const task = getTask(db, params.id)

    if (!task || task.project_id !== projectId) {
      return NextResponse.json({ error: 'Not found' }, { status: 404 })
    }

    const incomingDeps = getTaskDependencies(db, task.id, 'incoming')
    const outgoingDeps = getTaskDependencies(db, task.id, 'outgoing')

    const blockedBy = incomingDeps
      .map(dep => getTask(db, dep.depends_on_id))
      .filter((t): t is Task => t !== null)

    const blocking = outgoingDeps
      .map(dep => getTask(db, dep.task_id))
      .filter((t): t is Task => t !== null)

    return NextResponse.json({ blockedBy, blocking })
  } catch (e) {
    return NextResponse.json({ error: String(e) }, { status: 500 })
  }
}

export async function POST(
  req: Request,
  { params }: { params: { id: string } }
) {
  try {
    const db = getDb()
    const projectId = getProjectIdFromSession()
    const body = await req.json() as { dependsOnId: string }

    const task = getTask(db, params.id)
    if (!task || task.project_id !== projectId) {
      return NextResponse.json({ error: 'Not found' }, { status: 404 })
    }

    const depTask = getTask(db, body.dependsOnId)
    if (!depTask || depTask.project_id !== projectId) {
      return NextResponse.json({ error: 'Dependency not found' }, { status: 404 })
    }

    const { addDependency } = require('@/lib/db/taskDependencies')
    addDependency(db, params.id, body.dependsOnId)

    return NextResponse.json({ success: true })
  } catch (e) {
    return NextResponse.json({ error: String(e) }, { status: 400 })
  }
}
